	-reads stdin 
	-outputs in pipefd[0]

Intermediate odd : 
	-reads from pipefd[1]
	-outputs in pipefd[2]

Intermediate even : 
	-reads from pipedfd[3]
	-outputs in pipefd[0]

Last cmd : 
	-reads from pipefd[1] if odd - [3] if even
	-outputs to sdtdout


Odd cmd : 
		-reads from pipefd[1]
	if last cmd
		-outputs stdout
	else
		-outputs in pipefd[2]

Even  cmd : 
	if first cmd
		-reads stdin
	else
